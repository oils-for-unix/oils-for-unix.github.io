476 escape_arg() {
477 a="$1"
478 until [ -z "$a" ]; do
479 case "$a" in
480 (\'*) printf "'\"'\"'";;
481 (*) printf %.1s "$a";;
482 esac
483 a="${a#?}"
484 echo len=${#a} >&2
485 done
486 }
487
488 # encode
489 phrase="$(escape_arg "that's it!")"
490 echo escaped "$phrase"
491
492 # decode
493 eval "printf '%s\\n' '$phrase'"
494
495 echo ---

498 arg="$(printf ':\000:')"
499 #echo "arg=$arg"

503 *) echo escaped "$(escape_arg "$arg")" ;;
504 esac
