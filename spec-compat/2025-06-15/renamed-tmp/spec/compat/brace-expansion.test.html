158 #### Tilde expansion with brace expansion
159
160 # The brace expansion happens FIRST. After that, the second token has tilde
161 # FIRST, so it gets expanded. The first token has an unexpanded tilde, because
162 # it's not in the leading position.
163
164 HOME=/home/bob
165
166 # Command
167
168 echo {foo~,~}/bar
169
170 # Loop
171
172 for x in {foo~,~}/bar; do
173 echo -- $x
174 done
175
176 # Array
177
178 a=({foo~,~}/bar)
179
180 for y in "${a[@]}"; do
181 echo "== $y"
182 done
183
184 ## STDOUT:
185 foo~/bar /home/bob/bar
186 -- foo~/bar
187 -- /home/bob/bar
188 == foo~/bar
189 == /home/bob/bar
190 ## END
191
192 ## BUG mksh STDOUT:
193 foo~/bar ~/bar
194 -- foo~/bar
195 -- ~/bar
196 == foo~/bar
197 == ~/bar
198 ## END
