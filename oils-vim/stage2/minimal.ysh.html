#
# Single-quoted strings are raw by default
#

# Command mode
echo '/usr/bin'
echo r'C:\Program Files\'  # r'' prefix is allowed

# expression mode
var r1 = '/usr/bin/'
var r2 = r'C:\Program Files\' # r'' prefix is allowed

# corner cases for r''
echo ls --foo='bar'
echo ls --foo=r'bar'
echo ls bar'bar'
echo ls -r'bar'

# beginning of line
'echo'
r'echo'
u'echo'
$"echo"

#
# Double-quoted strings support interpolation (and a few \ escapes)
#

echo "hi $r1 \$ \" \\"
var dq = "hi $r2 \$ \" \\"

echo $"hi $r1 \$ \" \\"
var dq = $"hi $r2 \$ \" \\"

echo --foo=$"bar"

#
# J8 strings have \ escapes
#

echo b'hi \t \' \\'
var j8 = b'hi \t \' \\'

echo u'hi \t \' \\'
var j8 = u'hi \t \' \\'

#
# Triple-quoted strings
#  ''' r'''
#  """
#  b""" u"""
#

var empty = ''

# raw

echo '''
  $r1
  C:\Program Files\
  one ' 
  two ''
  ''' | cat

var x = '''
  C:\Program Files\
  one ' 
  two ''
  ''' ++ empty

echo r'''
  $r1
  C:\Program Files\
  one '
  two ''
  ''' | cat

var x = r'''
  C:\Program Files\
  one '
  two ''
  ''' ++ empty

# double quoted

echo """
  $r1
  one "
  two ""
  """

var x = """
  $r1
  one "
  two ""
  \""""

echo $"""
  $r1
  \$ one "
  \$ two ""
  \"""" other

var x = $"""
  $r1
  \$ one "
  \$ two ""
  """ ++ 'other'

  ''' | cat

  ''' ++ empty
